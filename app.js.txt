// Get the form and the list
const symptomForm = document.getElementById('symptomForm');
const symptomList = document.getElementById('symptomList');

// Listen for form submission
symptomForm.addEventListener('submit', function(event) {
    event.preventDefault(); // Prevent page reload on form submit

    const symptomInput = document.getElementById('symptom');
    const newSymptom = symptomInput.value; // Get the value from the input field

    if (newSymptom) {
        const listItem = document.createElement('li');
        listItem.textContent = newSymptom;
        symptomList.appendChild(listItem); // Add the symptom to the list

        symptomInput.value = ''; // Clear the input field
    }
});
